}); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON); } } function formatarData(d) { if (d instanceof Date) { return Utilities.formatDate(d, 'America/Sao_Paulo', 'yyyy-MM-dd'); } return String(d).trim(); } function verificarDisponibilidade(tipo, dataBusca) { try { if (!tipo || !dataBusca) { return ContentService.createTextOutput(JSON.stringify({ disponivel: true, vagas: 99 })).setMimeType(ContentService.MimeType.JSON); } const ss = SpreadsheetApp.openById(ID_SOLICITACOES); const aba = ss.getSheetByName(ABA_SOLICITACOES) || ss.getSheets()[0]; const rows = aba.getDataRange().getValues(); const grupoTipo = getTipoGrupo(tipo); const limite = LIMITES[grupoTipo] || 999; let count = 0; rows.slice(1).forEach(function(r) { if (!r[0] || !r[9]) return; const dataLinha = formatarData(r[9]); const tipoLinha = getTipoGrupo(String(r[1] || "")); const status = String(r[8] || "Pendente"); if (dataLinha === dataBusca && tipoLinha === grupoTipo && status !== 'Cancelado') { count++; } }); return ContentService.createTextOutput(JSON.stringify({ disponivel: count < limite, vagas: limite - count, limite: limite, agendados: count })).setMimeType(ContentService.MimeType.JSON); const aba = ss.getSheetByName(ABA_SOLICITACOES) || ss.getSheets()[0]; garantirCabecalho(aba); if (dados.data && dados.tipo) { const grupoTipo = getTipoGrupo(dados.tipo); const limite = LIMITES[grupoTipo]; if (limite !== undefined) { const rows = aba.getDataRange().getValues(); let count = 0; rows.slice(1).forEach(function(r) { if (!r[0] || !r[9]) return; const dataLinha = formatarData(r[9]); const tipoLinha = getTipoGrupo(String(r[1] || "")); const status = String(r[8] || "Pendente"); if (dataLinha === dados.data && tipoLinha === grupoTipo && status !== 'Cancelado') { count++; } }); if (count >= limite) { throw new Error('Limite atingido: ' + dados.tipo + ' já tem ' + count + '/' + limite + ' para ' + dados.data); } } } aba.appendRow([ new Date(), dados.tipo||"", dados.cliente||"", dados.comprador||"", dados.cidade||"", dados.servico||"", dados.residuo||"", dados.obs||"", "Pendente", dados.data||"" ]); } function atualizarStatus(linhaNum, novoStatus) { const ss = SpreadsheetApp.openById(ID_SOLICITACOES); const aba = ss.getSheetByName(ABA_SOLICITACOES) || ss.getSheets()[0]; aba.getRange(linhaNum, 9).setValue(novoStatus); } function garantirCabecalho(aba) { const pl = aba.getRange(1,1,1,10).getValues()[0]; if (pl[0] !== 'Data/Hora') { aba.getRange(1,1,1,10).setValues([['Data/Hora','Tipo Caçamba','Cliente','Comprador','Cidade','Serviço','Resíduo','Observação','Status','Data Solicitada']]); aba.getRange(1,1,1,10).setBackground('#1a5c2a').setFontColor('#ffffff').setFontWeight('bold'); } else if (pl[9] !== 'Data Solicitada') { aba.getRange(1,10).setValue('Data Solicitada'); } } function resposta(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function onOpen() { SpreadsheetApp.getUi().createMenu("🚛 Caçamba") .addItem("📋 Gerar Programação do Dia", "gerarProgramacaoDoDia") .addSeparator().addItem("📄 Exportar PDF", "exportarPDF").addToUi();
