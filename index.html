Para enviar dados direto ao Sheets, você precisa publicar um Web App no Apps Script.
Como criar o Web App →

Passo a passo para criar o Web App:

1. Acesse script.google.com
2. Crie um novo projeto e cole o código abaixo
3. Clique em Implantar → Nova implantação
4. Tipo: App da Web | Executar como: Eu | Acesso: Qualquer pessoa
5. Copie a URL gerada e cole no campo acima 
function doPost(e) {
  var sheet = SpreadsheetApp.openById('SEU_ID_AQUI')
    .getSheetByName('Solicitações') || 
    SpreadsheetApp.openById('SEU_ID_AQUI').getSheets()[0];
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([
    new Date(), data.comprador, data.tipo,
    data.servico, data.residuo, data.cliente,
    data.cidade, data.data, data.obs, 'Pendente'
  ]);
  return ContentService.createTextOutput('OK');
}
 
function doGet(e) {
  var sheet = SpreadsheetApp.openById('SEU_ID_AQUI')
    .getSheets()[0];
  var rows = sheet.getDataRange().getValues();
  var result = rows.slice(1).map(function(r) {
    return {
      timestamp: r[0], comprador: r[1], tipo: r[2],
      servico: r[3], residuo: r[4], cliente: r[5],
      cidade: r[6], data: r[7], obs: r[8], status: r[9]||'Pendente'
    };
  });
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
